\section*{Sets and Indices:}
% N = all hours, K = selected hedge instruments.
\begin{description}[widest={$K \subseteq F:$}]
  \item[$\bullet\; i \in N:$] hourly time index; $N$ is the set of all hours
  \item[$\bullet\; t \in T:$] time point $t$ of optimization in time $T$
  \item[$\bullet\; F:$] total set of hedge instruments, with
    \[
      \begin{aligned}
        F = \{&\mathrm{D^{Base}},\ \mathrm{WKND^{Base}},\ \mathrm{W^{Base}},\ \mathrm{M^{Base}},\ \mathrm{Q^{Base}},\ \mathrm{Y^{Base}},\\
            &\mathrm{D^{Peak}},\ \mathrm{WKND^{Peak}},\ \mathrm{W^{Peak}},\ \mathrm{M^{Peak}},\ \mathrm{Q^{Peak}},\ \mathrm{Y^{Peak}}\}
      \end{aligned}
    \]
  \item[$\bullet\; K \subseteq F:$] chosen subset of hedge instruments; $K$ is the set of selected hedge
    instruments from total set $F$
  \item[$\bullet\; j \in K:$] hedge instrument index; $j$ runs over the selected hedge instruments in
    $K$
  \item[$\bullet\; I^{j}:$] set of hours $i$ covered by hedge instrument
    $j$
\end{description}
  
\section*{Parameters:}  
% L_i, P_i^{spot}, and P_t^j are model inputs.
% epsilon_i controls the hourly tolerance band.
% H_min is the minimum hedge ratio of total load.
\begin{description}[widest={$\mathbf{1}_{\{i \in I^{j}\}}:$}]
  \item[$\bullet\; L_i:$] load in hour $i$ [MWh]
  \item[$\bullet\; P_i^{\text{spot}}:$] auction spot price in hour $i$ [EUR/MWh]
  \item[$\bullet\; P_t^{j}:$] forward price for hedge instrument $j$ at time $t$ [EUR/MWh]
  \item[$\bullet\; \mathbf{1}_{\{i \in I^{j}\}}:$] indicator function that equals 1 if hour $i$ is covered by hedge
    instrument $j$ and 0 otherwise, defined as
    \[
      \mathbf{1}_{\{i \in I^{j}\}} =
      \begin{cases}
        1, & i \in I^{j} \\
        0, & i \notin I^{j}
      \end{cases}
    \]
  \item[$\bullet\; \varepsilon_i:$] allowed relative coverage deviation in hour $i$; example: $\varepsilon_i = 0.05$
    for a 5\% tolerance
  \item[$\bullet\; H_{\min}:$] minimum hedge ratio of total load; example: $H_{\min} = 0.8$ for an 80\%
    hedge target
\end{description}

\section*{Decision Variables:}
% a^j is the hedge ratio decision for each selected instrument j.
\begin{description}[widest={$a^{j}:$}]
  \item[$\bullet\; a^{j}:$] hedge ratio $a$ for hedge instrument $j$
\end{description}

\section*{Objective Minimize Hedging Cost:}
% Simplified version:
% Minimize total hedging cost over all selected instruments and hours.
\[
  \min_{a^{j}} \; \sum_{j}^{K}\sum_{i}^{N} P_t^{j} \, a^{j} \mathbf{1}_{\{i \in I^{j}\}}L_i
\]

% \[
%   \min_{a^{j}} \; \sum_{j}^{K} {|I^{j}|} P_t^{j} \, a^{j} * \frac{\sum_{i}^{N} \mathbf{1}_{\{i \in I^{j}\}} L_i}{|I^{j}|}
% \]

\section*{Cost-Neutrality Condition:}
% Enforce that the hedged portfolio is cost-neutral against spot value.
\[
0 \mathrel{\stackrel{!}{=}} \sum_{i}^{N}
\Bigg(
  L_i
  - \sum_{j}^{K} \mathbf{1}_{\{i \in I^{j}\}} a^{j}
    \frac{1}{|I^{j}|}
    \sum_{h}^{N} \mathbf{1}_{\{h \in I^{j}\}} L_h
\Bigg)
P_i^{\text{spot}}
\]

  %  =

\section*{Coverage/Shape Constraints (Optional):}
% Keep the hedged hourly profile within a tolerance band around L_i.
\begin{align*}
  (1-\varepsilon_i)L_i \le \sum_{j}^{K} \mathbf{1}_{\{i \in I^{j}\}} \, a^{j} \frac{1}{|I^{j}|}
  \sum_{h}^{N} \mathbf{1}_{\{h \in I^{j}\}} L_h \le (1+\varepsilon_i)L_i,
  \quad \forall i \in N
\end{align*}

\section*{Bounds and Additional Conditions (Optional):}
% Bound hedge ratios and require at least H_min of total load to be hedged.
\begin{gather*}
-2 \le a^{j} \le 2, \quad \forall j \in K \\[0.8em]
\dfrac{\sum\limits_{j}^{K} a^{j} \sum\limits_{i}^{N} \mathbf{1}_{\{i \in I^{j}\}} L_i}{\sum\limits_{i}^{N} L_i} \ge H_{\min}
\end{gather*}
% Example: if a^j = 2.4 or a^j = -2.3 for any instrument j,
% the solution is infeasible.
% Allowed values are between -2 and 2, e.g., -1.2, 0, 0.8, or 2.0.

\section*{Final Model Selection:}
% Final combined optimization model assembled from the chosen constraints.
\begin{align} 
\min_{a^{j}} \quad & \sum_{j}^{K}\sum_{i}^{N} P_t^{j} \, a^{j} \, \mathbf{1}_{\{i \in I^{j}\}} L_i \tag{I}\\[0.7em]
\intertext{subject to:}
& 0 \mathrel{\stackrel{!}{=}} \sum_{i}^{N}\Bigg(L_i - \sum_{j}^{K} \mathbf{1}_{\{i \in I^{j}\}} a^{j}
  \frac{1}{|I^{j}|} \sum_{h}^{N} \mathbf{1}_{\{h \in I^{j}\}} L_h\Bigg)P_i^{\text{spot}} \tag{II}\\[0.7em]
& (1-\varepsilon_i)L_i \le \sum_{j}^{K} \mathbf{1}_{\{i \in I^{j}\}} \, a^{j} \frac{1}{|I^{j}|}
  \sum_{h}^{N} \mathbf{1}_{\{h \in I^{j}\}} L_h \le (1+\varepsilon_i)L_i,\quad \forall i \in N \tag{III}\\[0.7em]
& -2 \le a^{j} \le 2,\quad \forall j \in K \tag{IV}\\[0.7em]
& \dfrac{\sum\limits_{j}^{K} a^{j} \sum\limits_{i}^{N} \mathbf{1}_{\{i \in I^{j}\}} L_i}{\sum\limits_{i}^{N} L_i} \ge H_{\min} \tag{V}
\end{align}
